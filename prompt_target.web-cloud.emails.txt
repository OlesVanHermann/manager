### 11.10 DISPATCHER - DÃ‰TECTION AUTOMATIQUE

```typescript
// dispatcher.ts
export type EmailServiceType =
  | "exchange"       // Microsoft Exchange
  | "emailpro"       // Email Pro standard
  | "mxplan-modern"  // /email/mxplan (migrÃ©)
  | "mxplan-legacy"  // /email/domain (ancien)
  | "zimbra";        // Zimbra

export async function detectServiceType(serviceId: string): Promise<EmailServiceType> {
  // 1. Format org/service â†’ Exchange ou Zimbra
  if (serviceId.includes("/")) {
    const [org] = serviceId.split("/");
    if (org.includes("zimbra")) return "zimbra";
    return "exchange";
  }

  // 2. Tester EmailPro vs MXPlan
  try {
    await apiFetch(`/email/pro/${serviceId}/billingMigrated`);
    return "emailpro";
  } catch {
    // 3. MXPlan Modern ou Legacy
    try {
      await apiFetch(`/email/mxplan/${serviceId}`);
      return "mxplan-modern";
    } catch {
      return "mxplan-legacy";
    }
  }
}
```
