GET /hosting/web/{sn}/module/{id}          # Details module
POST /hosting/web/{sn}/module              # Installer module
DELETE /hosting/web/{sn}/module/{id}       # Supprimer module
POST /hosting/web/{sn}/module/{id}/changePassword # Changer mdp admin
