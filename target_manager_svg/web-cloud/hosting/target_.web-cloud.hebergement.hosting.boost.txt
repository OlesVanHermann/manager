# API ENDPOINTS

GET    /hosting/web/{serviceName}/boost
       -> Etat actuel du boost

POST   /hosting/web/{serviceName}/boost/activate
       -> Activer le boost
       Body: { duration: '1d' | '3d' | '7d' }

GET    /order/hosting/web/boost/{serviceName}
       -> Prix des options boost

POST   /order/hosting/web/boost/{serviceName}
       -> Commander un boost payant

# MODELE DONNEES

interface BoostStatus {
  active: boolean;
  expirationDate?: string;
  quotaRemaining: number;
  quotaTotal: number;
}

interface BoostOffer {
  duration: '1d' | '3d' | '7d';
  price: number;
  isFree: boolean;
  creditsRequired: number;
}

# REGLES METIER

- 3 boosts gratuits inclus par an (offres Pro/Performance)
- Boost = x2 ressources CPU/RAM
- Duree max 7 jours par activation
- Cumul impossible (1 boost actif max)
- Prise d'effet immediate
