# API ENDPOINTS

GET    /hosting/web/{serviceName}/cron
       -> Liste des taches planifiees

POST   /hosting/web/{serviceName}/cron
       -> Creer une nouvelle tache
       Body: { command, language, frequency, email, description }

GET    /hosting/web/{serviceName}/cron/{id}
       -> Details d'une tache

PUT    /hosting/web/{serviceName}/cron/{id}
       -> Modifier une tache

DELETE /hosting/web/{serviceName}/cron/{id}
       -> Supprimer une tache

# MODELE DONNEES

interface CronJob {
  id: number;
  command: string;
  language: 'php' | 'node' | 'python' | 'other';
  languageVersion?: string;
  frequency: string;      // Expression cron
  email?: string;
  state: 'enabled' | 'disabled';
  description?: string;
}

================================================================================

# REGLES METIER

- Minimum 5 minutes entre executions (*/5 minimum)
- Timeout selon offre (60s Perso, 300s Pro/Performance)
- Email = notification resultat execution
- PHP par defaut, Node.js disponible sur offres Pro+
- Syntaxe cron standard (minute heure jour mois jour_semaine)
- Maximum 10-50 taches selon offre
