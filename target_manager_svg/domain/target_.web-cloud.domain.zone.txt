| Colonne | Description |
|---------|-------------|

# API ENDPOINTS
GET /domain/zone/{zone}/record - Liste enregistrements
  Params: {
    fieldType: "A",
    subDomain: "www"
  }
  Response: [
    {
      id: 12345,
      fieldType: "A",
      subDomain: "",
      target: "203.0.113.50",
      ttl: 3600
    }
  ]

POST /domain/zone/{zone}/record - Créer enregistrement
  Body: {
    fieldType: "A",
    subDomain: "www",
    target: "203.0.113.50",
    ttl: 3600
  }
  Response: { id: 12346, taskId: 123456 }

PUT /domain/zone/{zone}/record/{id} - Modifier enregistrement
  Body: { target: "203.0.113.51", ttl: 7200 }
  Response: { taskId: 123457 }

DELETE /domain/zone/{zone}/record/{id} - Supprimer enregistrement
  Response: { taskId: 123458 }

POST /domain/zone/{zone}/refresh - Actualiser zone
  Response: { taskId: 123459 }
