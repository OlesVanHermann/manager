## 4. APIS & DONNEES

### Endpoints
```
GET /hosting/web/{serviceName}/module
POST /hosting/web/{serviceName}/module
DELETE /hosting/web/{serviceName}/module/{id}
POST /hosting/web/{serviceName}/module/{id}/changePassword
```

### Modele
```typescript
interface HostingModule {
  id: string;
  moduleId: string;
  name: string;
  domain: string;
  path: string;
  version: string;
  adminLogin: string;
  state: 'ok' | 'installing' | 'error';
}

interface AvailableModule {
  id: string;
  name: string;
  description: string;
  logo: string;
  versions: string[];
}
```
